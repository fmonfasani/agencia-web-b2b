   // Wrapper de fetch con X-API-Key automaticamente
   export const apiClient = {
     async call(url: string, options: RequestInit = {}) {
       const apiKey = localStorage.getItem('api_key');
       
       return fetch(url, {
         ...options,
         headers: {
           'Content-Type': 'application/json',
           ...(apiKey && { 'X-API-Key': apiKey }),
           ...options.headers
         }
       });
     },
     
     get: (url: string) => apiClient.call(url),
     post: (url: string, data: any) => 
       apiClient.call(url, { method: 'POST', body: JSON.stringify(data) }),
     // ... más métodos
   };

3. Crear hook lib/hooks/useApi.ts:

   export function useApi(baseUrl: string) {
     const [data, setData] = useState(null);
     const [loading, setLoading] = useState(false);
     const [error, setError] = useState(null);
     
     const fetch = async (endpoint: string, method = 'GET', body = null) => {
       setLoading(true);
       try {
         const response = await apiClient.call(`${baseUrl}${endpoint}`, {
           method,
           body: body ? JSON.stringify(body) : undefined
         });
         
         if (!response.ok) {
           if (response.status === 401) {
             // Redirigir a login
           }
           throw new Error(`API error: ${response.status}`);
         }
         
         const result = await response.json();
         setData(result);
         return result;
       } catch (err) {
         setError(err);
       } finally {
         setLoading(false);
       }
     };
     
     return { data, loading, error, fetch };
   }

   export function useAuth() {
     const [user, setUser] = useState(null);
     const apiKey = localStorage.getItem('api_key');
     
     useEffect(() => {
       if (!apiKey) return;
       
       // GET /tenant/me para obtener datos del usuario
       apiClient.call(`${SAAS_API_URL}/tenant/me`)
         .then(r => r.json())
         .then(setUser);
     }, [apiKey]);
     
     const login = async (email: string, password: string) => {
       const res = await apiClient.call(`${SAAS_API_URL}/auth/login`, {
         method: 'POST',
         body: JSON.stringify({ email, password })
       });
       
       const { api_key } = await res.json();
       localStorage.setItem('api_key', api_key);
       setUser(user);
     };
     
     const logout = () => {
       localStorage.removeItem('api_key');
       setUser(null);
     };
     
     return { user, apiKey, login, logout, isLoggedIn: !!apiKey };
   }
